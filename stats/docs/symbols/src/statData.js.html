100 	  if (value) {
101 	     if (!decimals) {
102 				 decimals = DECIMAL_COUNT;
103 	     }

106 	     return 0;

197 	function addElement(elem) {
198 		if (elem !== undefined && elem !== null) {
199 			if (onlyPositive && elem <= 0) {
200 				collectorWarning[collectorWarning.length] = elem;
201 			} else {
202 				totalCount++;
203 				if (calcAverageTemp) {
204 					averageTemp = round( (averageTemp + elem) / 2);
205 				} else {
206 					calcAverageTemp = true;
207 					averageTemp = elem;
208 				}
209 			}
210 		}

250 	* 	@function calculateMedian
251 	*		@private
252 	*		@return {number} median
253 	*/
254 	function calculateMedian() {
255 		var len = collector.length;
256 		var xm = 0;
257 		if (len > 0) {
258 			// vorher Array Werte sortieren !!!
259 			collector.sort();
260 			if (isOdd(len)) {
261 				var xn = (len + 1) / 2;
262 				console.log('isOdd xm: '+xn);
263 				xm = collector[xn];
264 			} else {
265 				var xm1 = (collector[(len / 2)]) / 2;
266 				var xm2 = (collector[((len + 2) /2 )]) / 2;
267 				console.log('isEven: xm1 '+xm1+' - xm2: '+xm2);
268 				xm = xm1 + xm2;
269 			}
270 		} else {
271 			console.warn('cannot calc median of empty array');
272 		}
273 
274 		return round(xm);
275 	};

297 	* 	@function calculateAverage
298 	*		@private
299 	*		@return {number} average
300 	*/
301 	function calculateAverage() {
302 		var total = 0.0;
303 		var temp = 0.0;
304 		var len = collector.length;
305 		var average = 0;
306 
307 		if (len > 0) {
308 			//console.log('wait for average... calc data ' + len);
309 			for (var i=(len-1); i >=0; i--) {
310 				//console.log('i: '+collector[name][i]);
311 				//temp += parseFloat(collector[i]);
312 				checkValue(total, collector[i]);
313 				total = round((total + collector[i]));
314 				//console.log(total);
315 				if (collector[i] > max) {
316 						max = collector[i];
317 				}
318 				if (collector[i] < min) {
319 						min = collector[i];
320 				}
321 			}
322 
323 			var newLen = len;
324 			if (calcAverageTemp) {
325 				newLen++;
326 			}
327 
328 			/*
329 			console.log('total: '+total);
330 			console.log('averageTemp: '+averageTemp);
331 			console.log('newLen: '+newLen);
332 			console.warn((total + averageTemp) / (newLen));
333 			*/
334 
335 			average = round((total + averageTemp) / (newLen));
336 
337 		} else {
338 			console.error('cannot find any collector or data for collector '+name);
339 		}
340 
341 		return (average);
342 	};
